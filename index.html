function addImagesToTrack(track) { const placed = []; // 配置済み画像の位置情報 images.forEach(src => { let tries = 0; let top, left, isOverlapping; do { isOverlapping = false; top = random(5, 75); left = random(5, 90); // 重なりチェック for (const pos of placed) { const dx = pos.left - left; const dy = pos.top - top; const distance = Math.sqrt(dx * dx + dy * dy); if (distance < 12) { // 近すぎると重なるのでやり直し isOverlapping = true; break; } } tries++; if (tries > 30) break; // 無限ループ防止 } while (isOverlapping); placed.push({ top, left }); const img = document.createElement('img'); img.src = src; img.className = 'hero-image'; img.style.top = `${top}%`; img.style.left = `${left}%`; const deg = random(-10, 10); img.style.setProperty('--rotate', `${deg}deg`); img.style.transform = `translateY(0px) rotate(${deg}deg)`; track.appendChild(img); }); }
